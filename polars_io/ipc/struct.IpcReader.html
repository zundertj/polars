Expand description

Read Arrows IPC format into a DataFrame

Example

use polars_core::prelude::*;
use std::fs::File;
use polars_io::ipc::IpcReader;
use polars_io::SerReader;

fn example() -> PolarsResult<DataFrame> {
    let file = File::open("file.ipc").expect("file not found");

    IpcReader::new(file)
        .finish()
}
